\begin{Eq*}
	\PVerb{\def\bin{\newbinaryop{bin}}}

\begin{Eq*}
	\explained{a \bin b} \\
\end{Eq*}

\begin{Eq*}
	\PVerb{\def\fn{\newfunc{fn}}}

	\explained{\fn(a, b)} \\
\end{Eq*}
